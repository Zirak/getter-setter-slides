              // Never use this function
              function prng() {
                let seed = 0xdeadcafe;
                return {
                  next() {
                    seed *= 1700;
                    seed %= 0x7fff;
                    return {
                      value: seed,
                      done: false,
                    };
                  },
                };
              }

              const rand = prng();
              rand.next(); // { value: 2563, done: false }
              rand.next(); // { value: 31856, done: false }

Aside: Implementing coroutines

  • Coroutines model a state machine
  • Progress it in next calls 

            function* fYield() {
              const x = yield;
              const y = yield x * 2;
              return y;
            }

            const fState = {
              [Symbol.iterator]() {
                let step = 0;
                let vars = {};
                return {
                  next(...args) {
                    switch (step) {
                      case 0:
                        step += 1;
                        return { value: undefined, done: false };
                      case 1:
                        step += 1;
                        vars.x = args[0];
                        return { value: vars.x * 2, done: false };
                      case 2:
                        step += 1;
                        vars.y = args[0];
                        return { value: vars.y, done: true };
                      case 3:
                        return { value: undefined, done: true };
                    }
                  }
                };
              },
            };

  • Values are lazy: Computed now
  • Functions are eager: Computed when called
Contrast to an IIFE: A function called now.
