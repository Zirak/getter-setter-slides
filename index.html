              [Symbol.iterator]() {
                let step = 0;
                let vars = {};
                return {
                  next(...args) {
                    switch (step) {
                      case 0:
                        step += 1;
                        return { value: undefined, done: false };
                      case 1:
                        step += 1;
                        vars.x = args[0];
                        return { value: vars.x * 2, done: false };
                      case 2:
                        step += 1;
                        vars.y = args[0];
                        return { value: vars.y, done: true };
                      case 3:
                        return { value: undefined, done: true };
                    }
                  }
                };
              },
