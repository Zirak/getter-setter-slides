              function iterate() {
                const arr = ['foo', 'bar'];
                let i = 0;
                return {
                  next() {
                    if (i >= arr.length) {
                      return { done: true };
                    }

                    const ret = arr[i];
                    i += 1;
                    return {
                      value: ret,
                      done: false,
                    };
                  },
                };
              }
            

              const items = iterate();
              items.next(); // { value: 'foo', done: false }
              items.next(); // { value: 'bar', done: false }
              items.next(); // { done: true }

Aside: Implementing coroutines

  • Coroutines model a state machine
  • Progress it in next calls 

            function* fYield() {
              const x = yield;
              const y = yield x * 2;
              return y;
            }

            const fState = {
              [Symbol.iterator]() {
                let step = 0;
                let vars = {};
                return {
                  next(...args) {
                    switch (step) {
                      case 0:
                        step += 1;
                        return { value: undefined, done: false };
                      case 1:
                        step += 1;
                        vars.x = args[0];
                        return { value: vars.x * 2, done: false };
                      case 2:
                        step += 1;
                        vars.y = args[0];
                        return { value: vars.y, done: true };
                      case 3:
                        return { value: undefined, done: true };
                    }
                  }
                };
              },
            };

Lazy vs. Eager


              const config = {
                apiPort: 4444,
              };

              // vs

              function getConfig() {
                return { apiPort: 4444 };
              }
  • Values are eager: Computed now
  • Functions are lazy: Computed when called
Contrast to an IIFE: A function called now.

Promises: Lazy vs. Eager


              const p = new Promise((resolve) => {
                console.log('ketchup');
                resolve(4);
              });
              console.log('mayo');
  • The resolver is called immediately
  • Promises are values in the future
